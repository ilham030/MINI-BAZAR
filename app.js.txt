// ============ QLOBAL DƏYİŞƏNLƏR ============
let PRODUCTS = [];
let CATEGORIES = [];
let cart = [];
let currentCategory = 'all';
let currentSearchTerm = '';

const CONFIG = {
    STORE_CURRENCY: 'AZN',
    STORE_PHONE: '994773911300',
};

// ============ İNİTİALİZASİYA ============
document.addEventListener('DOMContentLoaded', async function() {
    try {
        const response = await fetch('data.json?t=' + Date.now());
        const data = await response.json();
        
        PRODUCTS = data.products || [];
        CATEGORIES = data.categories || [];

        loadCartFromStorage();
        renderCategoryButtons();
        renderStoreInfo();

        setTimeout(() => {
            applyFilters();
        }, 400);

        window.addEventListener('scroll', handleScrollForBackToTop);

        document.addEventListener('keydown', function(e) {
            if (e.key === 'Escape') {
                const cartModal = document.getElementById('cart-modal');
                const productModal = document.getElementById('product-modal');
                if (productModal && !productModal.classList.contains('hidden')) {
                    closeProductModal();
                } else if (cartModal && !cartModal.classList.contains('hidden')) {
                    toggleCartModal();
                }
            }
        });
        
    } catch (error) {
        console.error("Məlumatları yükləmək mümkün olmadı:", error);
        const grid = document.getElementById('products-grid');
        if(grid) grid.innerHTML = '<p class="text-red-500 col-span-full text-center">Məlumatlar yüklənərkən xəta baş verdi. Zəhmət olmasa səhifəni yeniləyin.</p>';
    }
});

// ============ MAĞAZA MƏLUMATLARI VƏ UI ============
function renderStoreInfo() {
    const badge = document.getElementById('free-delivery-badge');
    if (badge) {
        badge.innerHTML = `<i class="fa-solid fa-truck mr-2"></i> Çatdırılma mövcuddur`;
    }
}

function renderCategoryButtons() {
    const container = document.getElementById('categories-container');
    if (!container) return;
    
    let html = `<button onclick="filterByCategory('all')" class="category-btn active px-4 py-2 bg-gray-200 rounded mr-2">Bütün Məhsullar</button>`;
    CATEGORIES.forEach(cat => {
        html += `<button onclick="filterByCategory('${cat.id}')" class="category-btn px-4 py-2 bg-gray-200 rounded mr-2">${cat.name}</button>`;
    });
    container.innerHTML = html;
}

function filterByCategory(categoryId) {
    currentCategory = categoryId;
    
    document.querySelectorAll('.category-btn').forEach(btn => {
        btn.classList.remove('bg-blue-500', 'text-white');
        btn.classList.add('bg-gray-200', 'text-black');
        if (btn.getAttribute('onclick').includes(categoryId)) {
            btn.classList.remove('bg-gray-200', 'text-black');
            btn.classList.add('bg-blue-500', 'text-white');
        }
    });
    applyFilters();
}

function applyFilters() {
    const grid = document.getElementById('products-grid');
    if (!grid) return;

    let filtered = PRODUCTS;

    if (currentCategory !== 'all') {
        filtered = filtered.filter(p => p.categoryId === currentCategory);
    }

    if (currentSearchTerm) {
        filtered = filtered.filter(p => p.name.toLowerCase().includes(currentSearchTerm.toLowerCase()));
    }

    renderProducts(filtered);
}

function renderProducts(productsToRender) {
    const grid = document.getElementById('products-grid');
    grid.innerHTML = '';

    if (productsToRender.length === 0) {
        grid.innerHTML = '<p class="text-gray-500 col-span-full text-center">Bu kateqoriyada məhsul tapılmadı.</p>';
        return;
    }

    productsToRender.forEach(product => {
        grid.innerHTML += `
            <div class="product-card border p-4 rounded-lg shadow hover:shadow-md transition-shadow">
                <img src="${product.image || 'https://via.placeholder.com/300'}" alt="${product.name}" class="w-full h-48 object-cover mb-4 rounded">
                <h3 class="text-lg font-bold">${product.name}</h3>
                <p class="text-gray-600 font-semibold my-2">${product.price} ${CONFIG.STORE_CURRENCY}</p>
                <button onclick="addToCart('${product.id}')" class="w-full bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded transition-colors">Səbətə At</button>
            </div>
        `;
    });
}

// ============ SƏBƏT MƏNTİQİ ============
function loadCartFromStorage() {
    const saved = localStorage.getItem('minibazar_cart');
    if (saved) {
        cart = JSON.parse(saved);
    }
    updateCartCount();
    renderCartItems();
}

function saveCartToStorage() {
    localStorage.setItem('minibazar_cart', JSON.stringify(cart));
    updateCartCount();
    renderCartItems();
}

function addToCart(productId) {
    const product = PRODUCTS.find(p => p.id == productId);
    if (!product) return;

    const existingItem = cart.find(item => item.id == product.id);
    if (existingItem) {
        existingItem.quantity += 1;
    } else {
        cart.push({ ...product, quantity: 1 });
    }
    saveCartToStorage();
    toggleCartModal();
}

function removeFromCart(productId) {
    cart = cart.filter(item => item.id != productId);
    saveCartToStorage();
}

function updateCartCount() {
    const countEl = document.getElementById('cart-count');
    if (countEl) {
        const totalCount = cart.reduce((sum, item) => sum + item.quantity, 0);
        countEl.textContent = totalCount;
    }
}

function renderCartItems() {
    const container = document.getElementById('cart-items-container');
    const totalEl = document.getElementById('cart-total');
    if (!container) return;

    if (cart.length === 0) {
        container.innerHTML = '<p class="text-center text-gray-500 py-4">Səbətiniz boşdur.</p>';
        if(totalEl) totalEl.textContent = `0 ${CONFIG.STORE_CURRENCY}`;
        return;
    }

    let html = '';
    let total = 0;

    cart.forEach(item => {
        const itemTotal = item.price * item.quantity;
        total += itemTotal;
        html += `
            <div class="flex justify-between items-center mb-3 border-b pb-3">
                <div>
                    <p class="font-bold">${item.name}</p>
                    <p class="text-sm text-gray-500">${item.price} ${CONFIG.STORE_CURRENCY} x ${item.quantity}</p>
                </div>
                <div class="flex items-center gap-3">
                    <p class="font-bold">${itemTotal} ${CONFIG.STORE_CURRENCY}</p>
                    <button onclick="removeFromCart('${item.id}')" class="text-red-500 text-sm hover:underline"><i class="fa-solid fa-trash"></i></button>
                </div>
            </div>
        `;
    });

    container.innerHTML = html;
    if (totalEl) totalEl.textContent = `${total} ${CONFIG.STORE_CURRENCY}`;
}

function toggleCartModal() {
    const modal = document.getElementById('cart-modal');
    if (modal) modal.classList.toggle('hidden');
}

function closeProductModal() {
    const modal = document.getElementById('product-modal');
    if (modal) modal.classList.add('hidden');
}

function handleScrollForBackToTop() {
    const btn = document.getElementById('back-to-top');
    if (btn) {
        if (window.scrollY > 300) btn.classList.remove('hidden');
        else btn.classList.add('hidden');
    }
}

// ============ WHATSAPP SİFARİŞ SİSTEMİ ============
function sendOrderToWhatsApp() {
    if (cart.length === 0) {
        alert("Səbətiniz boşdur! Sifariş vermək üçün məhsul əlavə edin.");
        return;
    }

    const nameInput = document.getElementById('customer-name');
    const addressInput = document.getElementById('customer-address');
    
    const name = nameInput ? nameInput.value.trim() : '';
    const address = addressInput ? addressInput.value.trim() : '';

    if (!name || !address) {
        alert("Zəhmət olmasa adınızı və çatdırılma ünvanınızı tam daxil edin.");
        return;
    }

    let message = `*Yeni Sifariş - MİNİBAZAR*\n\n`;
    message += `👤 *Müştəri:* ${name}\n`;
    message += `📍 *Ünvan:* ${address}\n\n`;
    message += `🛍️ *Sifarişlər:*\n`;

    let total = 0;
    cart.forEach((item, index) => {
        const itemTotal = item.price * item.quantity;
        total += itemTotal;
        message += `${index + 1}. ${item.name} - ${item.quantity} ədəd (${itemTotal} ${CONFIG.STORE_CURRENCY})\n`;
    });

    message += `\n💰 *Yekun Məbləğ:* ${total} ${CONFIG.STORE_CURRENCY}`;

    const whatsappUrl = `https://wa.me/${CONFIG.STORE_PHONE}?text=${encodeURIComponent(message)}`;
    window.open(whatsappUrl, '_blank');
}
